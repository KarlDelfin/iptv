npm install video.js @videojs/http-streaming

<template>
  <video ref="videoPlayer" class="video-js vjs-default-skin" controls></video>
</template>

<script setup>
import { onMounted, ref, onBeforeUnmount } from 'vue';
import videojs from 'video.js';
import 'video.js/dist/video-js.css';
import '@videojs/http-streaming'; // Enables M3U8/MPD support

const videoPlayer = ref(null);
let player = null;

onMounted(() => {
  player = videojs(videoPlayer.value, {
    sources: [{
      src: 'YOUR_URL_HERE.m3u8', // or .mpd
      type: 'application/x-mpegURL' // or 'application/dash+xml'
    }],
    fluid: true,
  });
});

onBeforeUnmount(() => {
  if (player) player.dispose();
});
</script>
